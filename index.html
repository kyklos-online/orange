s.bounceOffset = 0; // for sinusoidal bounce s.visible = false; slices.push(s); } // Tools const sword = new Sprite(assets.sword, 600, 100, {scale:0.6}); const racket = new Sprite(assets.racket, 600, 250, {scale:0.6});
