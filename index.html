``` ```js /** * Orange Game – ROK prototype * * Features: * • Orange rolls horizontally & wraps around. * • Orange slices bounce up/down. * • Smiley face drawn onto orange sprite. * • Orange sits on a cup (static image). * • Sword can “cut” orange into slices. * • Tennis racket can “smack” orange (adds a bounce impulse). * • Hand sprite lets you pick up / drag the orange. * • Rename the orange via UI. * • Free‑hand drawing on the orange surface. */ const canvas = document.getElementById('gameCanvas'); const ctx = canvas.getContext('2d'); // ---------- Asset loading ---------- const assets = { orange: loadImg('assets/orange.png'), slice: loadImg('assets/orange_slice.png'), sword: loadImg('assets/sword.png'), racket: loadImg('assets/racket.png'), hand: loadImg('assets/hand.png'), cup: loadImg('assets/cup.png') }; function loadImg(src) { const img = new Image(); img.src = src; return img; } // ---------- Helper ---------- function rectCollision(a, b) { return ( a.x < b.x + b.w && a.x + a.w > b.x && a.y < b.y + b.h && a.y + a.h > b.y ); } // ---------- Sprite base ---------- class Sprite { constructor(img, x, y, opts = {}) { this.img = img; this.x = x; this.y = y; this.scale = opts.scale || 1; this.w = img.width * this.scale; this.h = img.height * this.scale; this.vx = opts.vx || 0; this.vy = opts.vy || 0; this.visible = true; } draw() { if (!this.visible) return; ctx.drawImage(this.img, this.x, this.y, this.w, this.h); } update(dt) { this.x += this.vx * dt; this.y += this.vy * dt; } } // ---------- Game objects ---------- let orange = new Sprite(assets.orange, 100, 300, {scale: 0.6, vx: 80}); let cup = new Sprite(assets.cup, 80, 380, {scale: 0.7}); let sword = new Sprite(assets.sword, 600, 250, {scale: 0.5}); let racket = new Sprite(assets.racket, 400, 150, {scale: 0.5}); let hand = new Sprite(assets.hand, -200, -200, {scale: 0.4}); // off‑screen until used let slices = []; // will hold slice sprites after cut let isDrawing = false; // free‑hand drawing flag let drawPath = []; // points while drawing let dragging = false; // hand‑drag flag let nameTag = "Orange"; // default name // ---------- UI ---------- document.getElementById('renameBtn').addEventListener('click', () => { const txt = document.getElementById('orangeName').value.trim(); if (txt) nameTag = txt; }); // ---------- Input handling ---------- canvas.addEventListener('mousedown', (e) => { const mouse = getMousePos(e); if (rectCollision(mouse, orange)) { // inside orange → start drawing isDrawing = true; drawPath = [{x: mouse.x, y: mouse.y}]; } else if (rectCollision(mouse, hand)) { // clicking the hand icon (used to pick up) dragging = true; hand.offsetX = mouse.x - hand.x; hand.offsetY = mouse.y - hand.y; } }); canvas.addEventListener('mousemove', (e) => { const mouse = getMousePos(e); if (isDrawing) { drawPath.push({x: mouse.x, y: mouse.y}); } if (dragging) { hand.x = mouse.x - hand.offsetX; hand.y = mouse.y - hand.offsetY; orange.x = hand.x; // orange follows hand while dragging orange.y = hand.y; } }); canvas.addEventListener('mouseup', () => { isDrawing = false; dragging = false; }); canvas.addEventListener('mouseleave', () => { isDrawing = false; dragging = false; }); function getMousePos(evt) { const rect = canvas.getBoundingClientRect(); return { x: evt.clientX - rect.left, y: evt.clientY - rect.top, w: 1, h: 1 }; } // ---------- Game loop ---------- let lastTime = 0; function gameLoop(timestamp) { const dt = (timestamp - lastTime) / 1000; // seconds lastTime = timestamp; // ----- Update ----- orange.update(dt); // wrap‑around roll if (orange.x > canvas.width) orange.x = -orange.w; // bounce slices if they exist slices.forEach(s => { s.update(dt); // simple up‑and‑down sinusoid s.y += Math.sin(timestamp / 200) * 0.5; }); // ----- Interaction checks ----- // Sword cuts orange → create 3 slices if (rectCollision(sword, orange) && slices.length === 0) { for (let i = 0; i < 3; i++) { const slice = new Sprite( assets.slice, orange.x + i * 15, orange.y, {scale: 0.5, vy: -30} ); slices.push(slice); } orange.visible = false; // hide whole orange } // Racket smacks orange → give it a quick upward impulse if (rectCollision(racket, orange)) { orange.vy = -200; // after a short time gravity will pull it down again setTimeout(() => { orange.vy = 0; }, 300); } // Hand appears when mouse hovers over orange (optional visual cue) // (Here we simply keep it off‑screen unless you want a custom sprite) // ----- Render ----- ctx.clearRect(0, 0, canvas.width, canvas.height); // cup first (under orange) cup.draw(); // orange (or its slices) if (orange.visible) orange.draw(); slices.forEach(s => s.draw()); // other tools sword.draw(); racket.draw(); // draw free‑hand path on top of orange if (drawPath.length > 1) { ctx.strokeStyle = 'rgba(255,0,0,0.7)'; ctx.lineWidth = 3; ctx.beginPath(); ctx.moveTo(drawPath[0].x, drawPath[0].y); for (let i = 1; i < drawPath.length; i++) { ctx.lineTo(drawPath[i].x, drawPath[i].y); } ctx.stroke(); } // name tag (centered above orange) if (orange.visible) { ctx.font = '20px sans-serif'; ctx.fillStyle = '#222'; ctx.textAlign = 'center'; ctx.fillText(nameTag, orange.x + orange.w / 2, orange.y - 10); } requestAnimationFrame(gameLoop); } requestAnimationFrame(gameLoop); ```
